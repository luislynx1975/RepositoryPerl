SELECT CASE
          WHEN TIPO_CORRIDA = 'E' THEN '*'
       END AS EXTRAS,
       CAST( FECHA_HORA_CORRIDA AS DATE), R.DESTINO, (CAST(FECHA_HORA_CORRIDA AS TIME))AS HORA,
        S.ABREVIACION
FROM PDV_T_CORRIDA C INNER JOIN T_C_RUTA R ON C.ID_RUTA = R.ID_RUTA
                     INNER JOIN SERVICIOS S ON S.TIPOSERVICIO = C.TIPOSERVICIO
WHERE CAST(FECHA_HORA_CORRIDA AS DATE) = '2009-09-23' AND CAST(FECHA_HORA_CORRIDA AS TIME) > '00:00:00'
      AND R.DESTINO = 'ZAC'
ORDER BY HORA;
